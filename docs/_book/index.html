(defn coffee-has-milk [str]
  (boolean (some #(= str %) ["Latte" "American With Milk" "Cortado" "Cappuccino"])))
(defn coffee-has-chocolate [str]
  (boolean (some #(= str %) ["Cocoa" "Hot Chocolate"])))
